class ContaBancaria {


// 1. Bloqueado! Só a classe vê.

// 2. Acesso controlado (Porta de entrada)

fun depositar(valor: Double) {

// Validação de segurança

if (valor > 0) {

saldo += valor

println("Sucesso!")

}

}

}
