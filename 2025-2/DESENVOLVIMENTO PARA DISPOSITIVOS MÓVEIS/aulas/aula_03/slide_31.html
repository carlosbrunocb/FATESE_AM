class MainViewModel : ViewModel() {


// 1. Instanciamos o fornecedor de dados

private val repository = FrasesRepository()


// 2. Criamos a função que a tela (Activity) vai chamar

fun sortearNovaFrase(): String {

val frases = repository.getFrases()

return frases.random() // Mágica do Kotlin! ✨

}

}
