1. Objetivo

Este trabalho consiste em implementar a lógica (back-end) em JavaScript para a calculadora simples cujo front-end (HTML e CSS) já foi desenvolvido. A calculadora deve ser capaz de lidar com as quatro operações matemáticas básicas (+, -, *, /) e respeitar o uso de parênteses ( ) para definir a ordem das operações.

O desafio central é implementar um parser que entenda a precedência de operadores corretamente, e para isso, utilizaremos como base o algoritmo de Notação Polonesa Reversa (RPN).

2. O Desafio: A Ordem das Operações

Uma calculadora simples não pode simplesmente calcular da esquerda para a direita. Por exemplo, a expressão 2 + 3 * 4 não é (2 + 3) * 4 = 20, e sim 2 + (3 * 4) = 14. A multiplicação tem precedência sobre a adição.

O método mais robusto para resolver isso é transformar a expressão que digitamos (infixa) em uma fila que o computador entenda (pós-fixa ou RPN) e, em seguida, calcular essa fila.

Fundamentos: Notação Polonesa Reversa (RPN)

A RPN, ou notação pós-fixa, remove a ambiguidade e a necessidade de parênteses ao colocar o operador *depois* dos seus operandos.

  • 3 + 4 (Infixa) torna-se 3 4 + (RPN)
  • (3 + 4) * 2 (Infixa) torna-se 3 4 + 2 * (RPN)

Para implementar nossa calculadora, o processo tem duas etapas:

  1. Conversão (Algoritmo Shunting-Yard): Transforma a string infixa (ex: "(3+4)*2") em uma fila de tokens RPN (ex: ['3', '4', '+', '2', '*']).
  2. Avaliação (Calculadora RPN): Lê a fila RPN e usa uma pilha para calcular o resultado final.

Abaixo, apresentamos duas formas de implementar essa lógica em JavaScript. Para este trabalho, o Caminho 1 (DIY) é o foco principal, pois o objetivo é aprender o algoritmo. O Caminho 2 é apresentado como referência profissional.

Caminho 1: Implementação "Pura" em JavaScript (DIY)

Esta é a implementação direta dos algoritmos de Shunting-Yard e Avaliação RPN. É mais complexa, mas oferece um entendimento completo do processo.

/*
 * ==========================================================
 * ALGORITMO DE CALCULADORA RPN COMPLETO EM JAVASCRIPT
 * (Este código serve como exemplo de implementação)
 * ==========================================================
 */

/**
 * Função principal que o seu botão "=" deve chamar.
 * Recebe a string da expressão e retorna o resultado.
 * @param {string} expressionString - Ex: "2+(3*4)/9+50-(34/2)"
 * @returns {number|string} - O resultado ou uma mensagem de erro.
 */
function calcularExpressao(expressionString) {
    try {
        // 1. Tokenizar: Transforma a string em um array de "tokens"
        const tokens = tokenizar(expressionString);
        
        // 2. Converter: Transforma os tokens infixos em RPN (pós-fixos)
        const rpnTokens = infixToRpn(tokens);
        
        // 3. Avaliar: Calcula o resultado a partir dos tokens RPN
        const resultado = avaliarRpn(rpnTokens);
        
        return resultado;
    } catch (error) {
        console.error("Erro no cálculo:", error.message);
        return "Erro"; // Retorna 'Erro' para o visor da calculadora
    }
}

/**
 * Etapa 1: Tokenizar a expressão
 * Transforma "2.5+3" em ["2.5", "+", "3"]
 */
function tokenizar(expression) {
    // Regex para encontrar números (incluindo decimais) OU operadores/parênteses
    const regex = /(\d+\.?\d*|\.\d+|[+\-*/()])/g;
    const tokens = expression.match(regex);
    
    if (!tokens) {
        throw new Error("Expressão inválida");
    }
    return tokens;
}

/**
 * Etapa 2: Algoritmo Shunting-Yard (Infixo para RPN)
 */
function infixToRpn(tokens) {
    const filaSaida = [];
    const pilhaOperadores = [];
    
    // Precedência dos operadores
    const precedencia = {
        '*': 2,
        '/': 2,
        '+': 1,
        '-': 1
    };

    const isOperador = (token) => ['+', '-', '*', '/'].includes(token);
    const isNumero = (token) => !isNaN(parseFloat(token));

    for (const token of tokens) {
        if (isNumero(token)) {
            filaSaida.push(token);
        } 
        else if (isOperador(token)) {
            // Enquanto houver operador no topo da pilha com precedência maior ou igual
            while (
                pilhaOperadores.length > 0 &&
                isOperador(pilhaOperadores[pilhaOperadores.length - 1]) &&
                precedencia[pilhaOperadores[pilhaOperadores.length - 1]] >= precedencia[token]
            ) {
                filaSaida.push(pilhaOperadores.pop());
            }
            pilhaOperadores.push(token);
        } 
        else if (token === '(') {
            pilhaOperadores.push(token);
        } 
        else if (token === ')') {
            // Desempilha operadores para a saída até encontrar o "("
            while (pilhaOperadores.length > 0 && pilhaOperadores[pilhaOperadores.length - 1] !== '(') {
                filaSaida.push(pilhaOperadores.pop());
            }
            
            if (pilhaOperadores.length === 0) {
                throw new Error("Parênteses desbalanceados");
            }
            
            // Descarta o "("
            pilhaOperadores.pop();
        }
    }

    // Desempilha todos os operadores restantes
    while (pilhaOperadores.length > 0) {
        const op = pilhaOperadores.pop();
        if (op === '(') {
            throw new Error("Parênteses desbalanceados");
        }
        filaSaida.push(op);
    }
    
    return filaSaida;
}

/**
 * Etapa 3: Avaliar a expressão RPN
 */
function avaliarRpn(rpnTokens) {
    const pilha = [];

    for (const token of rpnTokens) {
        if (!isNaN(parseFloat(token))) {
            // Se for um número, empilha
            pilha.push(parseFloat(token));
        } else {
            // Se for um operador, desempilha os dois últimos números
            if (pilha.length < 2) {
                throw new Error("Expressão mal formada");
            }
            const b = pilha.pop();
            const a = pilha.pop();

            // Realiza a operação e empilha o resultado
            switch (token) {
                case '+':
                    pilha.push(a + b);
                    break;
                case '-':
                    pilha.push(a - b);
                    break;
                case '*':
                    pilha.push(a * b);
                    break;
                case '/':
                    if (b === 0) {
                        throw new Error("Divisão por zero");
                    }
                    pilha.push(a / b);
                    break;
                default:
                    throw new Error("Operador desconhecido: " + token);
            }
        }
    }

    if (pilha.length !== 1) {
        throw new Error("Expressão mal formada");
    }
    
    return pilha[0];
}

 * Função que o seu botão "=" deve chamar (VERSÃO COM BIBLIOTECA)

function calcularComBiblioteca(expressionString) {

4. Funcionalidade Extra: Histórico de Cálculos (Log)

Tarefa Bônus: Implementar um sistema de log que salva e exibe os cálculos anteriores.

4.1. Requisitos do Log

  • Cada cálculo bem-sucedido deve ser salvo.
  • O formato da entrada do log deve ser: [<Data> <hora>] <expressão> = <resultado>
    Exemplo: [24/10/2025 17:25] 2 + 3 = 5

4.2. Implementação (Backend) - Armazenamento

No desenvolvimento web, não podemos salvar um "arquivo local" no computador do usuário por razões de segurança. A ferramenta correta para isso é o localStorage do navegador, um pequeno banco de dados que persiste mesmo após fechar a aba.

Você deve criar uma função salvarNoLog(expressao, resultado) que será chamada pela sua função calcular(). 

function salvarNoLog(expressao, resultado) {
    // 1. Formatar a data e hora
    const agora = new Date();
    const dataFormatada = agora.toLocaleDateString('pt-BR'); // ex: "24/10/2025"
    const horaFormatada = agora.toLocaleTimeString('pt-BR', { 
        hour: '2-digit', 
        minute: '2-digit' 
    }); // ex: "17:34"

    // 2. Criar a string de log
    const logEntry = `[${dataFormatada} ${horaFormatada}] ${expressao} = ${resultado}`;

    // 3. Obter logs antigos do localStorage (se existirem)
    // Usamos JSON para salvar um array de strings, e não apenas uma string
    let logs = JSON.parse(localStorage.getItem('calculadora_logs')) || [];

    // 4. Adicionar o novo log (no início do array)
    logs.unshift(logEntry);

    // 5. Limitar a, por exemplo, 50 entradas (opcional)
    if (logs.length > 50) {
        logs.pop(); // Remove o item mais antigo
    }

    // 6. Salvar o array atualizado de volta no localStorage
    localStorage.setItem('calculadora_logs', JSON.stringify(logs));
}

/**
 * Mostra ou esconde o contêiner do log e o popula.
 */
function toggleLog() {
    const logContainer = document.getElementById('log-container');
    const btnToggleLog = document.getElementById('btn-toggle-log');

    const estaVisivel = logContainer.style.display === 'block';

    if (estaVisivel) {
        logContainer.style.display = 'none';
        btnToggleLog.textContent = 'Ver Histórico';
    } else {
        logContainer.style.display = 'block';
        btnToggleLog.textContent = 'Esconder Histórico';
        carregarLogs(); // Popula o log toda vez que é aberto
    }
}

/**
 * Lê os logs do localStorage e os exibe na lista 
      .
 */
function carregarLogs() {
    const logLista = document.getElementById('log-lista');
    const logs = JSON.parse(localStorage.getItem('calculadora_logs')) || [];

    // Limpa a lista antiga para não duplicar
    logLista.innerHTML = '';

    if (logs.length === 0) {
        logLista.innerHTML = '<li>Nenhum histórico salvo.</li>';
        return;
    }

    // Adiciona cada entrada de log como um item de lista (
    • )
    logs.forEach(entry => {
        const li = document.createElement('li');
        li.textContent = entry;
        logLista.appendChild(li);
    });
}
