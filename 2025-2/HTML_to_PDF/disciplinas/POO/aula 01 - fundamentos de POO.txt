class Circulo:
    def __init__(self, raio):
        self.raio = raio
    
    def calcular_area(self):
        return 3.14 * self.raio * self.raio
    
    def calcular_perimetro(self):
        return 2 * 3.14 * self.raio

# Criando um objeto
meu_circulo = Circulo(5)

# Chamando métodos do objeto
area = meu_circulo.calcular_area()
perimetro = meu_circulo.calcular_perimetro()

print(f"Área: {area:.2f}")
print(f"Perímetro: {perimetro:.2f}")

    def calcular_area(self):
        return self.lado ** 2
    
    def calcular_perimetro(self):

    def __str__(self):

<h2 class="text-2xl font-bold mb-3">O que é Encapsulamento?</h2>
<p class="text-lg mb-2">É o conceito de <span class="python-blue font-semibold">esconder detalhes de implementação</span> e proteger o acesso direto aos dados:</p>

<li>Evita alterações indevidas nos dados</li>
<li>Permite validação antes de modificações</li>

<div class="mb-3">
<p class="text-lg font-medium python-blue mb-1">Atributos Públicos</p>
<p class="text-lg">Acessíveis diretamente: <code>self.atributo</code></p>

<div class="mb-3">
<p class="text-lg font-medium python-blue mb-1">Atributos Protegidos</p>
<p class="text-lg">Convenção com um underscore: <code>self._atributo</code></p>

<div>
<p class="text-lg font-medium python-blue mb-1">Atributos Privados</p>
<p class="text-lg">Dois underscores (name mangling): <code>self.__atributo</code></p>

class ContaBancaria:
    def __init__(self, titular, saldo=0):
        self.titular = titular      # público
        self._tipo = "Corrente"     # protegido
        self.__saldo = saldo        # privado
        
    def depositar(self, valor):
        if valor &gt; 0:
            self.__saldo += valor
            return True
        return False
        
    def sacar(self, valor):
        if 0 &lt; valor &lt;= self.__saldo:
            self.__saldo -= valor
            return True
        return False
        
    def consultar_saldo(self):
        return self.__saldo

# Uso
conta = ContaBancaria("Maria", 1000)
print(conta.titular)        # Acesso permitido
print(conta._tipo)          # Acesso possível (convenção)
# print(conta.__saldo)      # Erro! Acesso negado
print(conta.consultar_saldo())  # 1000 (correto)
</pre>

class Temperatura:
    def __init__(self, celsius=0):
        self.__celsius = celsius
        
    @property
    def celsius(self):
        return self.__celsius
        
    @celsius.setter
    def celsius(self, valor):
        if valor &lt; -273.15:
            raise ValueError("Temperatura abaixo do zero absoluto!")
        self.__celsius = valor
        
    @property
    def fahrenheit(self):
        return self.__celsius * 9/5 + 32
        
    @fahrenheit.setter
    def fahrenheit(self, valor):
        self.celsius = (valor - 32) * 5/9

# Uso com properties
temp = Temperatura(25)
print(temp.celsius)      # 25 (acessa como atributo)
temp.celsius = 30        # Atribui valor com validação
print(temp.fahrenheit)   # 86.0 (cálculo automático)
temp.fahrenheit = 68     # Converte e atribui a celsius
print(temp.celsius)      # 20.0
</pre>

from abc import ABC, abstractmethod
import math

# Classe base abstrata
class Forma(ABC):
    def __init__(self, nome):
        self._nome = nome
    
    @property
    def nome(self):
        return self._nome
    
    @abstractmethod
    def calcular_area(self):
        pass
    
    def exibir_info(self):
        print(f"Forma: {self.nome}")
        print(f"Área: {self.calcular_area():.2f}")

# Classes derivadas
class Circulo(Forma):
    def __init__(self, raio):
        super().__init__("Círculo")
        self.__raio = raio
    
    def calcular_area(self):
        return math.pi * self.__raio ** 2

class Quadrado(Forma):
    def __init__(self, lado):
        super().__init__("Quadrado")
        self.__lado = lado
    
    def calcular_area(self):
        return self.__lado ** 2

def processar_formas(formas):
    for forma in formas:

        print("-" * 30)

# Lista com objetos de diferentes tipos
formas = [

# Mesmo método, comportamentos diferentes

<ul class="list-disc pl-6 text-lg space-y-1">
<li><code class="code-font bg-gray-100 px-1">__init__(self, ...)</code>: Inicializa o objeto</li>
<li><code class="code-font bg-gray-100 px-1">__str__(self)</code>: Representação textual do objeto</li>
<li><code class="code-font bg-gray-100 px-1">__repr__(self)</code>: Representação oficial (para depuração)</li>
<li><code class="code-font bg-gray-100 px-1">__eq__(self, other)</code>: Sobrecarga do operador ==</li>
<li><code class="code-font bg-gray-100 px-1">__len__(self)</code>: Implementa comportamento de len()</li>
</ul>
</div>
<div class="content-box pl-5 mb-6">
<h2 class="text-2xl font-bold mb-3">Classes Abstratas</h2>
<p class="text-lg mb-2">Permitem definir <span class="python-blue font-semibold">interfaces comuns</span> que classes derivadas devem implementar:</p>
<ul class="list-disc pl-6 text-lg space-y-1">
<li>Importação: <code class="code-font bg-gray-100 px-1">from abc import ABC, abstractmethod</code></li>
<li>Classe abstrata: <code class="code-font bg-gray-100 px-1">class MinhaClasse(ABC):</code></li>
<li>Método abstrato: <code class="code-font bg-gray-100 px-1">@abstractmethod def metodo(self):</code></li>
<li>Não podem ser instanciadas diretamente</li>
<li>Forçam implementação de métodos nas subclasses</li>
</ul>
</div>
<div class="content-box pl-5">
<h2 class="text-2xl font-bold mb-3">Tipos de Métodos</h2>
<div class="flex items-center mb-2">
<div class="w-4 h-4 rounded-full python-bg-yellow mr-2"></div>
<p class="text-lg"><span class="font-bold">Métodos de Instância:</span> Operam em atributos do objeto (self)</p>
</div>
<div class="flex items-center mb-2">
<div class="w-4 h-4 rounded-full python-bg-blue mr-2"></div>
<p class="text-lg"><span class="font-bold">Métodos de Classe:</span> <code class="code-font bg-gray-100 px-1">@classmethod</code> - Operam na classe (cls)</p>
</div>
<div class="flex items-center">
<div class="w-4 h-4 rounded-full bg-gray-400 mr-2"></div>
<p class="text-lg"><span class="font-bold">Métodos Estáticos:</span> <code class="code-font bg-gray-100 px-1">@staticmethod</code> - Funções utilitárias</p>
</div>
</div>
</div>

<div class="w-5/12 pl-8 flex flex-col justify-center">
<div class="bg-gray-100 p-5 rounded-lg code-font mb-6">
<div class="flex items-center mb-3">
<div class="w-3 h-3 rounded-full python-bg-blue mr-2"></div>
<p class="text-lg font-semibold">Exemplo de Métodos Especiais</p>
</div>
<pre class="bg-gray-800 text-white p-3 rounded text-sm overflow-auto">
class Retangulo:
    def __init__(self, largura, altura):
        self.largura = largura
        self.altura = altura
    
    def __str__(self):
        return f"Retângulo {self.largura}x{self.altura}"
    
    def __eq__(self, outro):
        return self.area() == outro.area()
    
    def area(self):
        return self.largura * self.altura

# Usando os métodos especiais
r1 = Retangulo(5, 3)
r2 = Retangulo(3, 5)
print(r1)           # __str__
print(r1 == r2)     # __eq__
</pre>
</div>

<p class="text-lg font-semibold">Classes Abstratas</p>

<pre class="bg-gray-800 text-white p-3 rounded text-sm overflow-auto">
from abc import ABC, abstractmethod

class Forma(ABC):
    @abstractmethod
    def calcular_area(self):
        pass
    
    @abstractmethod
    def calcular_perimetro(self):
        pass
    
class Quadrado(Forma):
    def __init__(self, lado):
        self.lado = lado
    
    def calcular_area(self):
        return self.lado ** 2
    
    def calcular_perimetro(self):
        return 4 * self.lado
</pre>
