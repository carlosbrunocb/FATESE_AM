// Obtendo referência ao elemento canvas e seu contexto 2D
const canvas = document.querySelector('canvas');
const ctx = canvas.getContext('2d');

// Definindo estilo e desenhando um retângulo preenchido
ctx.fillStyle = '#0ea5e9';  // Azul
ctx.fillRect(20, 20, 120, 80);  // x, y, largura, altura

// Definindo estilo e desenhando um círculo
ctx.strokeStyle = '#1f2937';  // Cinza escuro
ctx.lineWidth = 4;
ctx.beginPath();
ctx.arc(200, 60, 40, 0, Math.PI * 2);  // x, y, raio, ângulo inicial, ângulo final
ctx.stroke();

// Também podemos adicionar texto, gradientes, imagens, e animações
// ctx.drawImage(), ctx.createLinearGradient(), ctx.fillText(), etc.
