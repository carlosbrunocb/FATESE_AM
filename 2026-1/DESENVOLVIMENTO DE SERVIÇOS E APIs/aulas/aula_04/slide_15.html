Intermediação, Validação e Status Codes HTTP

src/controllers/tasks.controller.js
tasks.controller.js
01const service = require('../services/tasks.service');
02
03function getAll(req, res) {
04 return res.status(200).json(service.list());
05}
06
07function getById(req, res) {
08 const id = Number(req.params.id);
09 const task = service.findById(id);
10 // Validação: Recurso não encontrado
11 if (!task) {
12 return res.status(404).json({ msg: 'Task não encontrada' });
13 }
14 return res.status(200).json(task);
15}
16
17function create(req, res) {
18 const { title, done } = req.body;
19 // Validação de entrada (input validation)
20 if (!title) {
21 return res.status(400).json({ msg: 'Title é obrigatório' });
22 }
23 const newTask = service.create({ title, done });
24 return res.status(201).json(newTask);
25}
26
27module.exports = { getAll, getById, create };

Destaques da Implementação

Status 400: Bad Request Linha 21

Se o cliente enviar dados inválidos (ex: faltar o título), o controller deve bloquear e informar o erro antes de chamar a regra de negócio.

Status 404: Not Found Linha 12

Se o recurso solicitado pelo ID não existe, retornamos 404. É responsabilidade do Controller traduzir "null" do Service para uma resposta HTTP.

Status 201: Created Linha 24

Para criação de recursos (POST), utilizamos semanticamente o código 201 em vez do 200 genérico, confirmando que o recurso foi criado.
