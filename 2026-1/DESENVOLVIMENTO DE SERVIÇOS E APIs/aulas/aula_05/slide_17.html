Código: POST /login

Body Params + validação + Status Codes (200 / 400 / 401)

Express.js

req.body

Exemplo com validação

Envie email e senha no JSON e valide credenciais fixas.

Login

server.js

const CREDENCIAIS = { email: 'admin@teste.com', senha: '123456' }

// POST /login  (Content-Type: application/json)
app.post('/login', (req, res) => {
  const { email, senha } = req.body

  // 1) Validação do payload
  if (!email || !senha) {
    return res.status(400).json({ mensagem: 'Email e senha são obrigatórios.' })
  }

  // 2) Regra de negócio (credenciais fixas)
  const ok = email === CREDENCIAIS.email && senha === CREDENCIAIS.senha

  if (!ok) {
    return res.status(401).json({ mensagem: 'Credenciais inválidas.' })
  }

  return res.status(200).json({ mensagem: 'Acesso permitido.' })
})

Observação

Em um sistema real, não guarde senha em texto puro e use hash + banco de dados.
