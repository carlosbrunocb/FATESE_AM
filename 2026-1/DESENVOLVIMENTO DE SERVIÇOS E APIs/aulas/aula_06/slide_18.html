const validaLivro = (req, res, next) => {
// 1. Desestrutura dados do corpo da requisição
const { titulo, autor, ano } = req.body;
// 2. Verifica se algum campo obrigatório está faltando
if (!titulo || !autor || !ano) {
// 3. SHORT-CIRCUIT: Encerra com erro 400
return res.status(400).json({
erro: "Dados inválidos",
mensagem: "Os campos são obrigatórios."
});
}
// 4. Sucesso: Passa o controle para o Controller
next();
};
module.exports = validaLivro;

Sanitização de Entrada

Garante que o Controller só seja executado se os dados mínimos necessários (titulo, autor, ano) estiverem presentes no req.body.

Short-Circuit (Curto-Circuito)

Ao usar return res..., encerramos o ciclo Request-Response imediatamente. O next() nunca é chamado e a rota não continua.

Fluxo de Sucesso

Somente se todas as verificações passarem, a função next() é invocada, permitindo que a requisição chegue ao próximo passo.
