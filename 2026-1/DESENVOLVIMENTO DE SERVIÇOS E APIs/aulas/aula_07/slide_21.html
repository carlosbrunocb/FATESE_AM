Implementação do livroService.js com exportação XML e integração com fornecedor
Aula 07
Laboratório Prático
src/services/livroService.js
const xml2js = require('xml2js');
 
// Estado em memória (exemplo didático)
let livros = [
{ id: 1, titulo: '1984', autor: 'George Orwell', ano: 1949 },
{ id: 2, titulo: 'Fahrenheit 451', autor: 'Ray Bradbury', ano: 1953 }
];
// CRUD básico
const listarTodos = () => livros;
 
// Exportar dados em XML (Content Negotiation)
const listarTodosComoXML = () => {
const builder = new xml2js.Builder({ rootName: 'biblioteca', headless: true });
return builder.buildObject({ livro: livros });
};
 
// Simular consumo de Web Service legado (SOAP/XML)
const buscarLivrosFornecedor = async () => {
const xmlFornecedor = `<fornecedor><livro id="99"><titulo>O Guia do Mochileiro</titulo></livro></fornecedor>`;
const parser = new xml2js.Parser({ explicitArray: false });
try {
  const res = await parser.parseStringPromise(xmlFornecedor);
  return res.fornecedor.livro;
} catch (err) {
  throw new Error('Falha ao integrar com o fornecedor.');
}
};
 
module.exports = { listarTodos, listarTodosComoXML, buscarLivrosFornecedor };
Funções Principais
listarTodos: Retorna todos os livros
listarTodosComoXML: Converte para XML
buscarLivrosFornecedor: Integração SOAP
Pontos de Atenção
Importante: Sempre valide XML antes de processar e trate erros de parsing adequadamente.
Usar explicitArray: false para evitar o aninhamento desnecessário de objetos em arrays
