Laboratório Prático
src/controllers/livroController.js
const livroService = require('../services/livroService');
 
const listarLivros = (req, res, next) => {
try {
  res.set('Vary', 'Accept');
  const accept = req.headers['accept'] || '';
 
  // 406 se o cliente pedir um formato não suportado
  if (accept &&!(accept.includes('application/json') || accept.includes('application/xml'))) {
    return res.status(406).send('Not Acceptable');
  }
 
  // Negociação para XML
  if (accept.includes('application/xml')) {
    const xmlData = livroService.listarTodosComoXML();
    res.type('application/xml');
    return res.status(200).send(xmlData);
  }
 
  // Padrão: JSON
  const livros = livroService.listarTodos();
  return res.status(200).json(livros);
} catch (error) {
  next(error);
}
};
 
const buscarFornecedor = async (req, res, next) => {
try {
  const dadosConvertidos = await livroService.buscarLivrosFornecedor();
  return res.status(200).json({ origem: 'Fornecedor Legado', dados: dadosConvertidos });
} catch (error) {
  next(error);
}
};
 
module.exports = { listarLivros, buscarFornecedor };
Fluxo de Processamento
1
Leitura do Header: Controller lê Accept e define Content-Type
2
Negociação: Decide entre JSON ou XML baseado no Accept
3
Delegação: Chama serviço para conversão/formato
Validações
Importante: Controller define Vary: Accept para compatibilidade com caches
Retorna 406 quando formato não é suportado
