1 const { Sequelize, DataTypes } = require('sequelize');
2 const sequelize = new Sequelize('livraria_db', 'postgres', 'senha123', {
3 host: 'localhost', dialect: 'postgres', logging: false
4 });
5
6 const Livro = sequelize.define('Livro', {
7 id: { type: DataTypes.INTEGER, autoIncrement: true, primaryKey: true},
8 titulo: { type: DataTypes.STRING, allowNull: false },
9 autor: { type: DataTypes.STRING, allowNull: false },
10 ano: { type: DataTypes.INTEGER, allowNull: false }
11 }, { tableName: 'livros', timestamps: true });
12
13 // CRUD simplificado
14 async function exemploCRUD() {
15 await sequelize.sync();
16 const novo = await Livro.create({ titulo: '1984', autor: 'George Orwell', ano: 1949 });
17 const todos = await Livro.findAll();
18 const um = await Livro.findByPk(novo.id);
19 await um.update({ ano: 1950 });
20 }
