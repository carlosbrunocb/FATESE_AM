Implementação do Repository
// Arquivo: src/repositories/livroRepository.js
// Responsabilidade única: acesso a dados
const LivroModel = require('../models/Livro');
// Busca todos os livros
const buscarTodos = async () => {
try {
return await LivroModel.find();
} catch (error) {
throw new Error(`Erro ao buscar livros: ${error.message}`);
}
};
// Salva um novo livro
const salvar = async (dadosLivro) => {
try {
const novoLivro = new LivroModel(dadosLivro);
return await novoLivro.save();
} catch (error) {
throw new Error(`Erro ao salvar livro: ${error.message}`);
}
};
Funções Adicionais
// Busca livro por ID
const buscarPorId = async (id) => {
try {
return await LivroModel.findById(id);
} catch (error) {
throw new Error(`Livro não encontrado: ${error.message}`);
}
};
// Atualiza estoque
const atualizarEstoque = async (id, novaQuantidade) => {
try {
return await LivroModel.findByIdAndUpdate(
id,
{ estoque: novaQuantidade },
{ new: true }
);
} catch (error) {
throw new Error(`Erro ao atualizar estoque: ${error.message}`);
}
};
Responsabilidade Única
O Repository é a única camada que importa o ORM/ODM (Mongoose/Sequelize) e realiza operações de banco de dados. Se o banco mudar, apenas o repositório precisa ser reescrito.
