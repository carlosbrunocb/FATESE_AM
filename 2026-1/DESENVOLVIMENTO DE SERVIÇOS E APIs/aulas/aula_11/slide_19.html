src/services/livroService.js - Função realizarVenda
Regras de negócio: buscar, verificar, vender, atualizar
Implementação do Service
// Arquivo: src/services/livroService.js
const livroRepository = require('../repositories/livroRepository');
const realizarVenda = async (idLivro) => {
try {
// 1. Busca o livro pelo ID
const livro = await livroRepository.buscarPorId(idLivro);
if (!livro) {
throw new Error("Livro não encontrado.");
}
// 2. Verifica o estoque
if (livro.estoque <= 0) {
throw new Error("Operação negada: Livro fora de estoque.");
}
// 3. Aplica a lógica da venda
const novoEstoque = livro.estoque - 1;
// 4. Atualiza no repositório
const livroAtualizado = await livroRepository.atualizarEstoque(
idLivro, novoEstoque
);
return livroAtualizado;
} catch (error) {
throw error;
}
};
