Repositório de Usuários
Camada de acesso a dados para operações de usuário
src/repositories/usuarioRepository.js
Código do Repositório 
// src/repositories/usuarioRepository.js
const Usuario = require('../models/Usuario');

// Buscar usuário por email
const buscarPorEmail = async (email) => {
    return await Usuario.findOne({ email });
};

// Salvar novo usuário
const salvar = async (dadosUsuario) => {
    const novoUsuario = new Usuario(dadosUsuario);
    return await novoUsuario.save();
};

// Buscar usuário por ID (omitindo senha)
const buscarPorId = async (id) => {
    return await Usuario.findById(id).select('-senha');
};

// Atualizar usuário
const atualizar = async (id, dadosAtualizados) => {
    return await Usuario.findByIdAndUpdate(
        id, 
        dadosAtualizados, 
        { new: true, runValidators: true }
    ).select('-senha');
};

module.exports = { 
    buscarPorEmail, 
    salvar, 
    buscarPorId,
    atualizar 
};
Padrão Repository
O padrão Repository encapsula a lógica de acesso a dados, separando as operações de banco do domínio.
  • Abstração do MongoDB
  • Reutilização de código
  • Testabilidade
  • Separação de responsabilidades
