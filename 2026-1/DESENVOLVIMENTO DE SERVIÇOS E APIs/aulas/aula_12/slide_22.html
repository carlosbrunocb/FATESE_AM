Função registrar() 
// src/services/authService.js
const usuarioRepository = require('../repositories/usuarioRepository');
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');

const CHAVE_SECRETA = "MinhaChaveSuperSecretaDaLivraria123";

const registrar = async (nome, email, senha) => {
    // 1. Verificar se o e-mail já existe
    const usuarioExistente = await usuarioRepository.buscarPorEmail(email);
    if (usuarioExistente) {
        throw new Error("Este e-mail já está cadastrado.");
    }

    // 2. Criptografar a senha (Salt de 10 rounds)
    const senhaCriptografada = await bcrypt.hash(senha, 10);

    // 3. Salvar no banco via repositório
    const novoUsuario = await usuarioRepository.salvar({
        nome,
        email,
        senha: senhaCriptografada
    });

    return {
        id: novoUsuario._id,
        nome: novoUsuario.nome,
        email: novoUsuario.email,
        mensagem: "Usuário cadastrado com sucesso!"
    };
};

module.exports = { registrar, login, CHAVE_SECRETA };
Sobre o Registro
O serviço de autenticação centraliza a lógica de registro, validação e criptografia de senhas.
  • Verificação de email duplicado
  • Hash seguro com bcrypt
  • Retorno de dados limpos
  • Tratamento de erros
