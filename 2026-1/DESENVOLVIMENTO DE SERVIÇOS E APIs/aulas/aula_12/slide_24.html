Código do Controlador 
// src/controllers/authController.js
const authService = require('../services/authService');

// Controlador para cadastrar novo usuário
const cadastrarUsuario = async (req, res, next) => {
    try {
        const { nome, email, senha } = req.body;
        
        // Validação básica
        if (!nome || !email || !senha) {
            return res.status(400).json({ 
                erro: "Todos os campos são obrigatórios." 
            });
        }

        const novoUsuario = await authService.registrar(nome, email, senha);
        res.status(201).json({ 
            mensagem: "Usuário cadastrado com sucesso!", 
            id: novoUsuario._id 
        });
    } catch (error) {
        res.status(400).json({ erro: error.message });
    }
};

// Controlador para autenticar usuário (login)
const autenticarUsuario = async (req, res, next) => {
    try {
        const { email, senha } = req.body;
        
        const dadosLogin = await authService.login(email, senha);
        
        res.status(200).json(dadosLogin);
    } catch (error) {
        res.status(401).json({ erro: error.message });
    }
};

module.exports = { cadastrarUsuario, autenticarUsuario };
Sobre o Controlador
O controlador gerencia as requisições HTTP de autenticação, validando entradas e delegando a lógica para o serviço.
  • Validação de campos obrigatórios
  • Tratamento de erros com status HTTP
  • Integração com serviço de autenticação
  • Respostas JSON estruturadas
Fluxo de Requisição
1
Recebe requisição HTTP
2
Valida dados de entrada
3
Chama serviço de auth
4
Retorna resposta JSON
