Middleware de Proteção JWT

Validação de tokens JWT para proteção de rotas
src/middlewares/autenticaJWT.js
Código do Middleware 
// src/middlewares/autenticaJWT.js
const jwt = require('jsonwebtoken');
const { CHAVE_SECRETA } = require('../services/authService');

/**
 * Middleware para autenticação JWT
 */
const autenticaJWT = (req, res, next) => {
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1];

    if (!token) {
        return res.status(401).json({ 
            erro: "Acesso negado. Token não fornecido." 
        });
    }

    try {
        const dadosToken = jwt.verify(token, CHAVE_SECRETA);
        req.usuarioLogado = dadosToken;
        next();
    } catch (error) {
        return res.status(403).json({ 
            erro: "Token inválido ou expirado." 
        });
    }
};

module.exports = autenticaJWT;
Como Funciona
O middleware intercepta requisições, valida o token JWT e permite ou nega acesso às rotas protegidas.
  • Extrai token do header
  • Verifica assinatura
  • Valida expiração
  • Injeta dados do usuário
