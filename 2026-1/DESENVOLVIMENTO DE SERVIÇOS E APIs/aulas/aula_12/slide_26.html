Rotas de Autenticação

Definição das rotas para registro e login de usuários

src/routes/authRoutes.js
Código das Rotas 
// src/routes/authRoutes.js
const express = require('express');
const router = express.Router();
const authController = require('../controllers/authController');

// Rota para registro de novo usuário
// POST /api/auth/registrar
router.post('/auth/registrar', authController.cadastrarUsuario);

// Rota para autenticação (login)
// POST /api/auth/login
router.post('/auth/login', authController.autenticarUsuario);

// Rota para obter informações do usuário logado
// GET /api/auth/me
router.get('/auth/me', autenticaJWT, authController.getUsuarioLogado);

// Rota para renovar token (refresh token)
// POST /api/auth/refresh
router.post('/auth/refresh', authController.renovarToken);

module.exports = router;
