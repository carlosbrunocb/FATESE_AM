Middleware de Autenticação
// Middleware autenticaJWT.js
const autenticaJWT = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return res.status(401).json({ erro: 'Token não fornecido' });
  
  jwt.verify(token, CHAVE_SECRETA, (err, decoded) => {
    if (err) return res.status(401).json({ erro: 'Token inválido' });
    req.usuarioLogado = decoded;
    next();
  });
};
Middleware de Autorização
// Middleware autorizaRole.js
const autorizaRole = (rolesPermitidas) => {
  return (req, res, next) => {
    const usuario = req.usuarioLogado;
    if (!usuario) return res.status(401).json({ erro: 'Não autenticado' });
    
    if (!rolesPermitidas.includes(usuario.role)) {
      return res.status(403).json({ erro: 'Acesso negado' });
    }
    next();
  };
};
