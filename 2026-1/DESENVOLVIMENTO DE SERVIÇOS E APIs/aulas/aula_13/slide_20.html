Middleware de Autorização (RBAC)

Implementação do controle de acesso baseado em papéis
RBAC Ativo
Middleware autorizaRole.js
// Middleware de RBAC utilizando função de alta ordem (Closure)
const autorizaRole = (rolesPermitidas) => {
return (req, res, next) => {
// O req.usuarioLogado foi injetado pelo middleware anterior (autenticaJWT)
const usuario = req.usuarioLogado;
if (!usuario) {
return res.status(401).json({
erro: "Usuário não identificado."
});
}
// Verifica se o array de cargos permitidos inclui o cargo do usuário atual
if (!rolesPermitidas.includes(usuario.role)) {
// Status 403 Forbidden: Eu sei quem você é, mas você não tem permissão.
return res.status(403).json({
erro: "Acesso Negado. Seu perfil não tem permissão para realizar esta ação."
});
}
next(); // Usuário autorizado, segue em frente!
};
};
module.exports = autorizaRole;
Importante
Este middleware deve ser aplicado após o autenticaJWT para garantir que req.usuarioLogado exista.
