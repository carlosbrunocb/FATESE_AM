// Importação do Mongoose
const mongoose = require('mongoose');

// Definição do Schema do Livro
const LivroSchema = new mongoose.Schema({
    titulo: {
        type: String,
        required: true
    },
    autor: {
        type: String,
        required: true
    },
    ano: {
        type: Number,
        required: true
    },
    estoque: {
        type: Number,
        default: 0
    }
}, { timestamps: true });

// Exportação do modelo
module.exports = mongoose.model('Livro', LivroSchema);
