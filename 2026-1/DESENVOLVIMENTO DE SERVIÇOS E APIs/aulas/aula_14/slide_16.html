// Importação do repositório
const livroRepository = require('../repositories/livroRepository');

// Listar todos os livros (com filtro opcional por título)
const listarTodos = async (titulo) => {
    return await livroRepository.buscarTodos(titulo);
};

// Criar novo livro com validação de ano
const criar = async (dadosLivro) => {
    const anoAtual = new Date().getFullYear();
    
    // Validação: ano não pode estar no futuro
    if (dadosLivro.ano > anoAtual) {
        throw new Error("O ano de lançamento não pode estar no futuro.");
    }
    
    return await livroRepository.salvar(dadosLivro);
};

// Exportação dos serviços
module.exports = { listarTodos, criar };
