// Importação do serviço
const livroService = require('../services/livroService');

// Controller para listar livros
const listarLivros = async (req, res, next) => {
    try {
        // Captura query parameter: /api/livros?titulo=Express
        const { titulo } = req.query;
        const livros = await livroService.listarTodos(titulo);
        res.status(200).json(livros);
    } catch (error) {
        next(error);
    }
};

// Controller para criar livro
const criarLivro = async (req, res, next) => {
    try {
        const novoLivro = await livroService.criar(req.body);
        res.status(201).json(novoLivro);
    } catch (error) {
        res.status(400).json({ erro: error.message });
    }
};

module.exports = { listarLivros, criarLivro };
