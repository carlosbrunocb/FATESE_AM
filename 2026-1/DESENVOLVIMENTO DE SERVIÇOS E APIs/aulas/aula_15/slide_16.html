Exemplo de Implementação
Código de configuração das rotas
app.js - Configuração de Rotas
const express = require('express');
const rotasV1 = require('./routes/v1/livroRoutes');
const rotasV2 = require('./routes/v2/livroRoutes');

const app = express();
app.use(express.json());

// Injeção de Rotas Versionadas
app.use('/api/v1', rotasV1);
app.use('/api/v2', rotasV2);
