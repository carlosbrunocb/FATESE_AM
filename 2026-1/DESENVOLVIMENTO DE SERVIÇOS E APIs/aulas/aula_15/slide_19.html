Implementar busca com paginação

const LivroModel = require('../models/Livro');

// Buscar todos com paginação
const buscarTodos = async (titulo, pagina = 1, limite = 0) => {
  let filtro = {};
  if (titulo) {
    filtro.titulo = { $regex: titulo, $options: 'i' };
  }
  const pular = (pagina - 1) * limite;
  return await LivroModel.find(filtro).skip(pular).limit(limite);
};

// Contar total de registros
const contarTodos = async (titulo) => {
  let filtro = {};
  if (titulo) filtro.titulo = { $regex: titulo };
  return await LivroModel.countDocuments(filtro);
};
module.exports = { buscarTodos, contarTodos };
Explicação Detalhada
Como funciona a paginação
Por que paginação?
A paginação evita retornar milhares de registros de uma vez, reduzindo carga no banco de dados e melhorando performance da API.
Parâmetros:
• pagina: número da página atual
• limite: quantidade de itens por página
• skip: quantos registros pular
Cálculo do Skip
skip = (pagina - 1) × limite
Página 1: skip 0 | Página 2: skip 10 | Página 3: skip 20
