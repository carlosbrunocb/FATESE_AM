const livroRepository = require('../../repositories/livroRepository');

// Service V1: Mantém compatibilidade total
const listarTodosV1 = async (titulo) => {
// Passamos limite 0 para trazer todos os registros
// Mantendo comportamento exato da V1 original
return await livroRepository.buscarTodos(titulo, 1, 0);
};

// Exportação do serviço V1
module.exports = {
listarTodosV1
};

Por que manter V1?
O serviço V1 preserva o comportamento original da API, garantindo que clientes existentes continuem funcionando sem necessidade de atualização.
Características:
• Sem paginação: Retorna array completo
• Formato simples: Array plano de objetos
• Sem metadados: Apenas dados brutos
