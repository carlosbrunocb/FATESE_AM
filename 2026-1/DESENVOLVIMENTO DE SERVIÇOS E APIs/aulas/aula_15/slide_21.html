Implementar lógica de paginação
livroService.js
JavaScript
const livroRepository = require('../../repositories/livroRepository');

const listarTodosV2 = async (titulo, pagina = 1, limite = 10) => {
  const livros = await livroRepository.buscarTodos(titulo, pagina, limite);
  const total = await livroRepository.contarTodos(titulo);
  
  // V2 retorna objeto com metadados
  return {
    dados: livros,
    paginacao: {
      totalRegistros: total,
      paginaAtual: Number(pagina),
      totalPaginas: Math.ceil(total / limite)
    }
  };
};

module.exports = { listarTodosV2 };
Nova Estrutura de Resposta
Formato objeto com metadados
Por que mudar para objeto?
A V2 retorna um objeto estruturado com dados e metadados de paginação, permitindo ao cliente saber quantas páginas existem.
Estrutura da resposta:
• dados: array de livros
• paginacao: metadados (total, página, páginas)
• totalPaginas: calculado com Math.ceil
Cálculo de Páginas
totalPaginas = Math.ceil(total / limite)
Ex: 50 registros / 10 por página = 5 páginas
