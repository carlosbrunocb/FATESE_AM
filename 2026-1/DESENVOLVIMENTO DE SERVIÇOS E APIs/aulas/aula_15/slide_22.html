const livroServiceV1 = require('../../services/v1/livroService');

const listarLivros = async (req, res, next) => {
    try {
        // Adicionando header de aviso de depreciação
        res.set('Deprecation', 'true');
       
        const livros = await livroServiceV1.listarTodosV1(req.query.titulo);
        res.status(200).json(livros); // Retorna ARRAY
    } catch (error) {
        next(error);
    }
};
module.exports = { listarLivros };
⚠️ Importante: Header Deprecation: true avisa que versão será descontinuada

Controller com nova estrutura
livroController.js (v2)
JavaScript
const livroServiceV2 = require('../../services/v2/livroService');

const listarLivros = async (req, res, next) => {
    try {
        const { titulo, pagina, limite } = req.query;
        const resultado = await livroServiceV2.listarTodosV2(
            titulo, pagina, limite
        );
        res.status(200).json(resultado); // Retorna OBJETO
    } catch (error) {
        next(error);
    }
};
module.exports = { listarLivros };
Diferença Principal
V1 retorna array simples, V2 retorna objeto com metadados de paginação
