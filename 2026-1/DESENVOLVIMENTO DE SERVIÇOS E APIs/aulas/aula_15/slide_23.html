JavaScript
const express = require('express');
const router = express.Router();
const controllerV1 = require('../../controllers/v1/livroController');

// Rota para listar todos os livros (V1)
router.get('/livros', controllerV1.listarLivros);

// Rota para buscar livro por ID (V1)
router.get('/livros/:id', controllerV1.buscarLivro);

// Rota para criar livro (V1)
router.post('/livros', controllerV1.criarLivro);

module.exports = router;
