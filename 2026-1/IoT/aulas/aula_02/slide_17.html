semaforo_delay.ino

// Pinos dos LEDs
const int ledVerde = 19;
const int ledAmarelo = 18;
const int ledVermelho = 5;

void setup() {
  pinMode(ledVerde, OUTPUT);
  pinMode(ledAmarelo, OUTPUT);
  pinMode(ledVermelho, OUTPUT);
}

void loop() {
  // Fase 1: Verde (Aberto)
  digitalWrite(ledVerde, HIGH);
  digitalWrite(ledAmarelo, LOW);
  digitalWrite(ledVermelho, LOW);
  delay(3000); // Para tudo por 3 segundos

  // Fase 2: Amarelo (Atenção)
  digitalWrite(ledVerde, LOW);
  digitalWrite(ledAmarelo, HIGH);
  delay(1000); // Para tudo por 1 segundo

  // Fase 3: Vermelho (Fechado)
  digitalWrite(ledAmarelo, LOW);
  digitalWrite(ledVermelho, HIGH);
  delay(3000); // Para tudo por 3 segundos
}

Problema: Durante cada delay(), o ESP32 fica bloqueado e não consegue executar outras tarefas!
