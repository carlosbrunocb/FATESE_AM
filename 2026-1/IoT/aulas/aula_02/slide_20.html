// Pinos dos LEDs
const int ledVerde = 19;
const int ledAmarelo = 18;
const int ledVermelho = 5;

// Variáveis para controle de tempo e estado
unsigned long tempoAnterior = 0;
int estadoSemaforo = 0; // 0=Verde, 1=Amarelo, 2=Vermelho

// Tempos de cada fase (em milissegundos)
const long intervaloVerde = 3000;
const long intervaloAmarelo = 1000;
const long intervaloVermelho = 3000;

void setup() {
  pinMode(ledVerde, OUTPUT);
  pinMode(ledAmarelo, OUTPUT);
  pinMode(ledVermelho, OUTPUT);
}

Destaque: Variáveis de controle de tempo (unsigned long) e estado (int) permitem temporização sem bloquear o processador!
