void loop() {
  unsigned long tempoAtual = millis(); // Checa o relógio do sistema

  // Lógica de Máquina de Estados
  if (estadoSemaforo == 0) { // Estado Verde
    digitalWrite(ledVerde, HIGH);
    digitalWrite(ledVermelho, LOW);
    
    // Verifica se já passou o tempo do Verde
    if (tempoAtual - tempoAnterior >= intervaloVerde) {
      tempoAnterior = tempoAtual; // Reseta o cronômetro
      estadoSemaforo = 1;         // Vai para o próximo estado
      digitalWrite(ledVerde, LOW);
    }
  } 
  else if (estadoSemaforo == 1) { // Estado Amarelo
    digitalWrite(ledAmarelo, HIGH);
    
    if (tempoAtual - tempoAnterior >= intervaloAmarelo) {
      tempoAnterior = tempoAtual;
      estadoSemaforo = 2;
      digitalWrite(ledAmarelo, LOW);
    }
  }

Observe: O loop não fica travado! A cada iteração, verificamos o tempo decorrido e atualizamos o estado conforme necessário.
