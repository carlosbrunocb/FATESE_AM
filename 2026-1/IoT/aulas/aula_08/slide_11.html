Código Fonte
1// Inclui a biblioteca WiFi do ESP32
2#include "WiFi.h"
3
4void setup() {
5 Serial.begin(115200); // Inicializa serial
6 WiFi.mode(WIFI_STA); // Define como estação
7 WiFi.disconnect(true); // Garante desconexão
8 delay(100); // Estabiliza o rádio
9}
10
11void loop() {
12 Serial.println("Escaneando redes...");
13 int n = WiFi.scanNetworks();
14
15 if (n == 0) {
16 Serial.println("Nenhuma rede encontrada.");
17 } else {
18 Serial.printf("%d redes encontradas:\n", n);
19 for (int i = 0; i < n; ++i) {
20 Serial.printf("%d: %s (%d dBm)\n", i+1,
21 WiFi.SSID(i).c_str(), WiFi.RSSI(i));
22 }
23 }
24 delay(5000);
25}
