Validando o Sistema como um todo

tests/test_sistema.py

def test_sistema_fluxo_completo(tmp_path):

# 1. Setup: Ambiente Real Controlado

arquivo_log = tmp_path / "historico.txt"

logger = Logger(caminho=str(arquivo_log))

app = CalculadoraApp(logger=logger)


# 2. Action: Simular uso real

app.iniciar()

app.somar(1, 1); app.somar(2, 2); app.somar(3, 3)


# 3. Verification: Validar saídas do sistema

relatorio = app.abrir_relatorio_logs()

assert "1 + 1 = 2" in relatorio

assert "2 + 2 = 4" in relatorio

assert "3 + 3 = 6" in relatorio
