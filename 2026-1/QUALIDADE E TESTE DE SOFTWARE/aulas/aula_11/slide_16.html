import pytest
from src.calculadora import somar, dividir, multiplicar

# Testes de funcionalidades existentes (Segurança de Regressão)
@pytest.mark.regressao
def test_soma_positiva():
    assert somar(2, 3) == 5

@pytest.mark.regressao
def test_multiplicacao():
    assert multiplicar(10, 5) == 50

# Teste que irá falhar devido ao Bug de Regressão (Soma Negativa)
def test_soma_negativa():
    assert somar(-2, -3) == -5

# Teste que irá falhar e precisa de correção (Cenário de Erro)
def test_divisao_por_zero():
    # O objetivo é que o código retorne None ou mensagem em vez de quebrar
    assert dividir(10, 0) == "Erro: Divisão por zero"
Testes que Passam
test_soma_positiva e test_multiplicacao - Funcionalidades básicas que funcionam corretamente, marcadas com @pytest.mark.regressao
Testes que Falham
test_soma_negativa - Falha devido a bug de regressão na função somar
test_divisao_por_zero - Falha devido a ZeroDivisionError
