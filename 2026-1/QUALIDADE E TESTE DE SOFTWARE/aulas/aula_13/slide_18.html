# 1. Importar bibliotecas necessárias from selenium import webdriver from selenium.webdriver.common.by import By # 2. Inicializar o WebDriver (Chrome) driver = webdriver.Chrome() try: # 3. Abrir a página web driver.get("https://example.org") # 4. Ler e imprimir o título da página print("Título da página:", driver.title) # 5. Ler a URL atual print("URL atual:", driver.current_url) # 6. Obter o código-fonte da página page_source = driver.page_source print("Tamanho do HTML:", len(page_source), "caracteres") finally: # 7. Fechar o navegador driver.quit()
